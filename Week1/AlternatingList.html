First version of the alternating function#

def const(n):
    mylist = []
    for i in range(n):
        mylist.append(3)
    return mylist

const(5)

[3, 3, 3, 3, 3]

def alt1(n):
    mylist = const(n)
    return mylist

def alt1(n):
    mylist = const(n)
    for i in range(1,n,2):
        mylist[i] = 8
    return mylist

def alt2(n):
    mylist = []
    for i in range(n):
        if i%2 == 0:
            mylist.append(3)
        else:
            mylist.append(8)
    return mylist

def alt2(n):
    mylist = []
    for i in range(n):
        if i%2 == 0:
            mylist.append(3)
        elif i%2 == 1:
            mylist.append(8)
    return mylist
